// This runs on the server for each visit
export async function getServerSideProps(context) {
  const { req } = context
  const authToken = req.cookies.authToken
  
  // If user isn't logged in, redirect to login page
  if (!authToken) {
    return {
      redirect: {
        destination: '/login',
        permanent: false,
      },
    }
  }
  
  try {
    // Get user info and notifications in parallel (faster!)
    const [userResponse, notificationsResponse] = await Promise.all([
      fetch(`https://api.myapp.com/user`, {
        headers: { Authorization: `Bearer ${authToken}` }
      }),
      fetch(`https://api.myapp.com/notifications`, {
        headers: { Authorization: `Bearer ${authToken}` }
      })
    ])
    
    const user = await userResponse.json()
    const notifications = await notificationsResponse.json()
    
    return {
      props: {
        user,
        notifications
      }
    }
  } catch (error) {
    // If something goes wrong, show error page
    return {
      props: {
        user: null,
        notifications: [],
        error: 'Unable to load dashboard data'
      }
    }
  }
}
