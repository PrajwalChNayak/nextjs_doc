// pages/api/users.js
export default function handler(req, res) {
  const { method } = req

  switch (method) {
    case 'GET':
      // Get all users
      res.status(200).json({ 
        users: ['John', 'Jane', 'Bob'],
        message: 'Users retrieved successfully! ğŸ“‹'
      })
      break
      
    case 'POST':
      // Create a new user
      const { name } = req.body
      res.status(201).json({ 
        message: `User ${name} created! ğŸ‰`,
        user: { id: Date.now(), name }
      })
      break
      
    case 'DELETE':
      // Delete a user
      res.status(200).json({ 
        message: 'User deleted! ğŸ—‘ï¸'
      })
      break
      
    default:
      res.setHeader('Allow', ['GET', 'POST', 'DELETE'])
      res.status(405).end(`Method ${method} Not Allowed`)
  }
}
